'use client';

import { useEffect, useRef } from 'react';
import gsap from 'gsap';

export default function HeroBanner() {
  // ✅ Sirf 1 ref kaafi hai pure section ke liye
  const containerRef = useRef(null);

  useEffect(() => {
    // gsap.context() Next.js mein animations manage karne ka sabse safe tareeqa hai
    const ctx = gsap.context(() => {
      // 1 ref se andar ke elements target karna
      const heading = containerRef.current.querySelector('.banner-heading');
      const subtitle = containerRef.current.querySelector('.banner-subtitle');

      if (!heading || !subtitle) return;

      // Text ko characters mein split karna
      const originalText = heading.textContent;
      heading.innerHTML = '';
      
      const chars = originalText.split('').map((char) => {
        const span = document.createElement('span');
        span.textContent = char === ' ' ? '\u00A0' : char;
        span.style.display = 'inline-block';
        heading.appendChild(span);
        return span;
      });

      // Characters animation (load par play hogi)
      gsap.from(chars, {
        yPercent: 50,
        opacity: 0,
        stagger: 0.03,
        duration: 0.85,
        ease: 'power3.out',
        delay: 0.2,
      });

      // Subtitle animation
      gsap.from(subtitle, {
        y: 25,
        opacity: 0,
        duration: 0.6,
        delay: 0.6,
        ease: 'power2.out',
      });

    }, containerRef); // Context is scoped to this section

    // ✅ Route change ya component unmount par automatically cleanup
    return () => ctx.revert();
  }, []); // [] = sirf 1 baar run hoga (page load par)

  return (
    <section ref={containerRef} className="banner-section">
      <div className="banner-wrapper">
        <h1 className="banner-heading">
          We Build Digital Products That Matter
        </h1>
        <p className="banner-subtitle">
          Innovative design, clean code, and seamless user experiences.
        </p>
      </div>

      <style>{`
        .banner-section {
          min-height: 100vh;
          display: flex;
          align-items: center;
          justify-content: center;
          background: #080808;
          color: #ffffff;
          padding: 2rem;
          overflow: hidden;
        }
        .banner-wrapper {
          max-width: 900px;
          text-align: center;
        }
        .banner-heading {
          font-size: clamp(2.2rem, 5vw, 4.5rem);
          font-weight: 800;
          line-height: 1.1;
          margin: 0;
          overflow: hidden; /* Animation clip ke liye zaroori */
        }
        .banner-subtitle {
          margin-top: 1.5rem;
          font-size: clamp(1rem, 2vw, 1.3rem);
          color: #a3a3a3;
          line-height: 1.6;
        }
      `}</style>
    </section>
  );
}
